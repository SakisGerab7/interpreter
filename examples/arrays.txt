let arr = [1, 2, 3, 4, 5];

disp arr;

arr = arr.map(fn(n) -> 2 * n);

disp "Double:";
disp arr;

arr = arr.filter(fn(n) -> n >= 5);

disp "Greater or equal to 5:";
disp arr;

let sum = arr.reduce(fn(acc, n) -> acc + n, 0);

disp "Sum of elements = " + sum;

let str = arr.reduce(fn(acc, n, i) -> acc + "arr[" + i + "] = " + n + " | ", " | ");

disp str;
